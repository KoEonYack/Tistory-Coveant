``` python
from collections import deque

def solution(cacheSize, cities):
    cache = deque()
    time = 0

    if cacheSize != 0:
        for city in cities:
            city = city.lower()
            if city in cache:
                cache.remove(city)
                cache.appendleft(city)
                time += 1
            elif city not in cache:
                if len(cache) >= cacheSize:
                    cache.pop()
                    cache.appendleft(city)
                    time += 5
                else:
                    cache.appendleft(city)
                    time += 5
    else:
        time += len(cities) * 5

    return time
```
