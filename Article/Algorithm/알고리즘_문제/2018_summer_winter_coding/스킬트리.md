def solution(skill, skill_trees):
    ans = 0

    for skill_tree in skill_trees:
        tmp = []
        for ch in skill_tree:
            if ch in skill:
                tmp.append(ch)

        flag = True
        for i in range(len(tmp)):
            if tmp[i] != skill[i]:
                flag = False
                break
        if flag is True:
            ans += 1

    return ans
